# Frontend
cd frontend
npm run build
gcloud builds submit --tag gcr.io/capstone-cicd-ai/frontend .
gcloud run deploy frontend --image gcr.io/capstone-cicd-ai/frontend --region us-central1 --allow-unauthenticated
cd ..

# Webhook Handler

cd ..

# Orchestrator
